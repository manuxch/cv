\section*{Áreas de especialización e intereses de investigación}

\begin{itemize}
	\item \textbf{Fluidos y materia blanda:} propiedades estructurales y dinámicas de fluidos confinados, transporte en nanocanales, pervaporación.
	\item \textbf{Materiales granulares:} descarga de silos, atascamientos, transporte, dinámica de \textit{stick-slip}.
	\item \textbf{Biofísica:} Modelado de tejidos biológicos, interacciones proteína-lípido.
	\item \textbf{Modelado computacional y simulación:} Dinámica Molecular, Monte Carlo, DEM (Método de Elementos Discretos), ABM (Modelado Basado en Agentes), Dinámica de Sistemas, Optimización Estocástica (Algoritmo Genético, \textit{Particle Swarm Optimization}).
\end{itemize}
